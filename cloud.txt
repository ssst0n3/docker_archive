#cloud-config
user: root
password: root
chpasswd: {expire: False}
ssh_pwauth: True
write_files:
- content: |
    #!/bin/bash
    sed -i s/1/0/g /etc/apt/apt.conf.d/20auto-upgrades
    sync
  path: /tmp/disable-auto-upgrades.sh
  permissions: '0755'
- content: |
    #!/bin/bash
    echo "PermitRootLogin yes" >> /etc/ssh/sshd_config
    ssh-keygen -A
    sync
  path: /tmp/sshd-config.sh
  permissions: '0755'
- content: |
    #!/bin/bash
    apt purge -y snapd
    apt autoremove -y
    apt clean
    sync
  path: /tmp/clean.sh
  permissions: '0755'
- content: |
    #!/bin/bash
    set -ex
    /tmp/disable-auto-upgrades.sh
    /tmp/sshd-config.sh
    /tmp/clean.sh
    echo "init.sh complete"
    sync
  path: /tmp/init.sh
  permissions: '0755'
runcmd:
  - /tmp/init.sh

  - [ write-files, always ]

  - [ runcmd, always ]
  - [ set-passwords, always ]
